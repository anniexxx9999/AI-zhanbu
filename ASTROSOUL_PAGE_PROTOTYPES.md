  useEffect(() => {
    const eventSource = new EventSource('/api/interpret-spouse?chartId=xxx');

        setIsStreaming(false);

    };

    <div className="report-content">
      <ReactMarkdown>{displayedText}</ReactMarkdown>
      {isStreaming && <span className="cursor-blink">|</span>}
    </div>
  );
};
